# hooks/useSwipeGesture.ts
export function useTabSwipe(currentTab, tabs, setTab) {
  return useDrag(({ swipe: [swipeX] }) => {
    const idx = tabs.indexOf(currentTab);
    if (swipeX < 0 && idx < tabs.length - 1) {
      setTab(tabs[idx + 1]);
    } else if (swipeX > 0 && idx > 0) {
      setTab(tabs[idx - 1]);
    }
  });
}
